
import React, { useState, useEffect } from "react";
import { User, PoliceInteraction, KnowYourRightsModule } from "@/entities/all";
import { Shield, BookOpen, FileText, AlertTriangle, TrendingUp, Users, Plus, Target, Zap, LogIn } from "lucide-react";
import { Link } from "react-router-dom";
import { createPageUrl } from "@/utils";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Badge } from "@/components/ui/badge";
import { Alert, AlertDescription, AlertTitle } from "@/components/ui/alert";

import StatsCard from "../components/dashboard/StatsCard";
import RecentIncidents from "../components/dashboard/RecentIncidents";
import RightsProgress from "../components/dashboard/RightsProgress";

export default function Dashboard() {
  const [interactions, setInteractions] = useState([]);
  const [rightsModules, setRightsModules] = useState([]);
  const [isLoading, setIsLoading] = useState(true);
  const [isAuthenticated, setIsAuthenticated] = useState(false);

  useEffect(() => {
    const checkAuthAndLoadData = async () => {
        try {
            await User.me();
            setIsAuthenticated(true);
            await loadDashboardData();
        } catch (error) {
            // User is not logged in or session expired
            setIsAuthenticated(false);
            console.log("User not authenticated, showing public welcome page.", error);
        } finally {
            setIsLoading(false);
        }
    };
    checkAuthAndLoadData();
  }, []);

  const loadDashboardData = async () => {
    try {
      const [interactionsData, modulesData] = await Promise.all([
        PoliceInteraction.list("-created_date", 10),
        KnowYourRightsModule.list("priority_level", 5)
      ]);
      setInteractions(interactionsData);
      setRightsModules(modulesData);
    } catch (error) {
      console.error("Error loading dashboard data:", error);
    } 
    // finally is handled in checkAuthAndLoadData for the overall loading state
  };

  const getStatsData = () => {
    const totalInteractions = interactions.length;
    const seriousIncidents = interactions.filter(i => 
      i.severity_level === 'serious' || i.severity_level === 'severe'
    ).length;
    const complaintsNeeded = interactions.filter(i => 
      i.follow_up_needed && !i.complaint_filed
    ).length;
    const recentInteractions = interactions.filter(i => {
      const interactionDate = new Date(i.created_date);
      const thirtyDaysAgo = new Date();
      thirtyDaysAgo.setDate(thirtyDaysAgo.getDate() - 30);
      return interactionDate >= thirtyDaysAgo;
    }).length;

    return {
      totalInteractions,
      seriousIncidents, 
      complaintsNeeded,
      recentInteractions
    };
  };

  const stats = getStatsData();

  // Show welcome content if user is not logged in OR is a new user
  const showWelcomeView = !isAuthenticated || interactions.length === 0;

  if (showWelcomeView) {
    return (
      <div className="p-6 space-y-8">
        <div className="max-w-6xl mx-auto">
          {/* Welcome Header - Updated for corruption victims */}
          <div className="text-center mb-8">
            <div className="w-16 h-16 bg-gradient-to-br from-amber-500 to-amber-600 rounded-xl flex items-center justify-center mx-auto mb-4">
              <img 
                src="https://qtrypzzcjebvfcihiynt.supabase.co/storage/v1/object/public/base44-prod/public/82b755008_image.png" 
                alt="Due Process AI"
                className="w-10 h-10 object-contain brightness-0 invert"
              />
            </div>
            <h1 className="text-3xl font-bold text-slate-900 mb-2">You Are Not Alone</h1>
            <p className="text-lg text-slate-600 max-w-3xl mx-auto">
              If you've had to leave home to escape a corrupt system, if you can't find proper legal representation, 
              if it feels like the whole world is against you - <strong>this platform was built for you</strong>.
            </p>
          </div>
          
          {!isAuthenticated && (
              <div className="text-center mb-8">
                  <Button size="lg" className="bg-amber-600 hover:bg-amber-700" onClick={() => User.login()}>
                      <LogIn className="w-5 h-5 mr-2" />
                      Sign Up or Log In to Get Started
                  </Button>
              </div>
          )}

          {/* Corruption Victim Support Banner */}
          <Alert className="bg-red-50 border-red-200 mb-8">
            <Shield className="h-5 w-5 text-red-600" />
            <AlertTitle className="text-red-900 text-lg">For Those Forced to Flee</AlertTitle>
            <AlertDescription className="text-red-800 mt-2">
              Whether you're facing malicious prosecution, judicial corruption, prosecutorial misconduct, or systematic retaliation - 
              you have constitutional rights that cannot be taken away. This platform gives you the tools to fight back and reclaim your life.
            </AlertDescription>
          </Alert>

          {/* Emergency/Crisis Actions */}
          <div className="grid grid-cols-1 lg:grid-cols-3 gap-8 mb-8">
            <Card className="border-0 shadow-lg bg-gradient-to-br from-red-50 to-red-100 hover:shadow-xl transition-all border-l-4 border-l-red-600">
              <CardContent className="p-6 text-center">
                <div className="w-12 h-12 bg-red-600 rounded-xl flex items-center justify-center mx-auto mb-4">
                  <AlertTriangle className="w-6 h-6 text-white" />
                </div>
                <h3 className="text-xl font-semibold text-slate-900 mb-3">Document Everything</h3>
                <p className="text-slate-600 mb-4">
                  Start by securely documenting the corruption, misconduct, or retaliation you've experienced.
                </p>
                <Link to={createPageUrl("ReportIncident")}>
                  <Button className="w-full bg-red-600 hover:bg-red-700">
                    Report Corruption/Misconduct
                  </Button>
                </Link>
                <Badge className="mt-3 bg-red-600 text-white">Critical First Step</Badge>
              </CardContent>
            </Card>

            <Card className="border-0 shadow-lg bg-gradient-to-br from-purple-50 to-purple-100 hover:shadow-xl transition-all border-l-4 border-l-purple-600">
              <CardContent className="p-6 text-center">
                <div className="w-12 h-12 bg-purple-600 rounded-xl flex items-center justify-center mx-auto mb-4">
                  <Zap className="w-6 h-6 text-white" />
                </div>
                <h3 className="text-xl font-semibold text-slate-900 mb-3">Build Your Case</h3>
                <p className="text-slate-600 mb-4">
                  Use AI to analyze evidence and build a strategy to expose the corruption and clear your name.
                </p>
                <Link to={createPageUrl("SelfLitigantCenter")}>
                  <Button className="w-full bg-purple-600 hover:bg-purple-700">
                    AI Case Strategy
                  </Button>
                </Link>
                <Badge className="mt-3 bg-purple-100 text-purple-700 border-purple-200">AI Powered</Badge>
              </CardContent>
            </Card>

            <Card className="border-0 shadow-lg bg-gradient-to-br from-blue-50 to-blue-100 hover:shadow-xl transition-all border-l-4 border-l-blue-600">
              <CardContent className="p-6 text-center">
                <div className="w-12 h-12 bg-blue-600 rounded-xl flex items-center justify-center mx-auto mb-4">
                  <Users className="w-6 h-6 text-white" />
                </div>
                <h3 className="text-xl font-semibold text-slate-900 mb-3">Find Others</h3>
                <p className="text-slate-600 mb-4">
                  Connect with others who faced similar corruption. Strength in numbers.
                </p>
                <Link to={createPageUrl("ClassActionRegistry")}>
                  <Button className="w-full bg-blue-600 hover:bg-blue-700">
                    Join the Registry
                  </Button>
                </Link>
              </CardContent>
            </Card>
          </div>

          {/* Survival Message */}
          <Card className="border-0 shadow-lg bg-gradient-to-r from-slate-900 to-amber-900 text-white mb-8">
            <CardContent className="p-8 text-center">
              <h2 className="text-2xl font-bold mb-4">The Path Back Home</h2>
              <p className="text-slate-300 text-lg leading-relaxed mb-4">
                "The arc of the moral universe is long, but it bends toward justice." - Martin Luther King Jr.
              </p>
              <p className="text-amber-200 font-medium">
                You were forced to leave to survive. Now we help you build the case to return with justice.
              </p>
            </CardContent>
          </Card>

          {/* Feature Overview */}
          <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-6">
            {[
              {
                title: "Secure Documentation",
                description: "Safely document and store evidence of rights violations",
                icon: Shield,
                color: "text-green-600"
              },
              {
                title: "AI Analysis", 
                description: "Get intelligent analysis of your evidence and legal situation",
                icon: Zap,
                color: "text-purple-600"
              },
              {
                title: "Community Support",
                description: "Connect with others who have similar experiences", 
                icon: Users,
                color: "text-blue-600"
              },
              {
                title: "Legal Resources",
                description: "Access templates, guides, and educational materials",
                icon: FileText,
                color: "text-amber-600"
              }
            ].map((feature, index) => (
              <Card key={index} className="border-0 shadow-sm bg-white text-center p-6">
                <feature.icon className={`w-8 h-8 ${feature.color} mx-auto mb-4`} />
                <h3 className="font-semibold text-slate-900 mb-2">{feature.title}</h3>
                <p className="text-sm text-slate-600">{feature.description}</p>
              </Card>
            ))}
          </div>
        </div>
      </div>
    );
  }

  // Returning user dashboard
  return (
    <div className="p-6 space-y-8">
      <div className="max-w-7xl mx-auto">
        {/* Header */}
        <div className="mb-8">
          <div className="flex items-center gap-3 mb-4">
            <div className="w-12 h-12 bg-gradient-to-br from-amber-500 to-amber-600 rounded-xl flex items-center justify-center">
              <img 
                src="https://qtrypzzcjebvfcihiynt.supabase.co/storage/v1/object/public/base44-prod/public/82b755008_image.png" 
                alt="Due Process AI"
                className="w-8 h-8 object-contain brightness-0 invert"
              />
            </div>
            <div>
              <h1 className="text-3xl font-bold text-slate-900">Welcome Back</h1>
              <p className="text-slate-600 mt-1">Empowering your pro se journey: track your progress, manage documented events, and identify next steps.</p>
            </div>
          </div>
        </div>

        {/* Stats Grid */}
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-6 mb-8">
          <StatsCard
            title="Battles Documented"
            value={stats.totalInteractions}
            icon={FileText}
            bgColor="bg-blue-500"
            description="Documented instances of due process violations."
          />
          <StatsCard
            title="Violations Uncovered"
            value={stats.seriousIncidents}
            icon={AlertTriangle}
            bgColor="bg-red-500"
            description="Incidents revealing procedural failures."
          />
          <StatsCard
            title="Next Strategic Moves"
            value={stats.complaintsNeeded}
            icon={TrendingUp}
            bgColor="bg-amber-600"
            description="Proactive steps for legal engagement."
          />
          <StatsCard
            title="Recent Engagements"
            value={stats.recentInteractions}
            icon={Users}
            bgColor="bg-slate-600"
            description="Activity in the past 30 days."
          />
        </div>

        {/* Main Content Grid */}
        <div className="grid lg:grid-cols-3 gap-8">
          <div className="lg:col-span-2 space-y-8">
            <RecentIncidents 
              interactions={interactions}
              isLoading={isLoading}
            />
          </div>

          <div className="space-y-8">
            <RightsProgress 
              modules={rightsModules}
              isLoading={isLoading}
            />
          </div>
        </div>
      </div>
    </div>
  );
}
